1. package main
  3. import (
  4. "machine"
  6. // The bot's code
  7. "github.com/conejoninja/go-escornabot/bot"
  8. "github.com/conejoninja/go-escornabot/input"
  10. "tinygo.org/x/drivers/buzzer" // buzzer driver
  11. "tinygo.org/x/drivers/easystepper" // motor driver
  12. )
  14. func main() {
  15. motors := easystepper.NewDual(machine.D0, machine.D7, machine.D1, machine.D9, machine.D10, machine.D12, machine.D11, machine.D13, 200, 60)
  16. motors.Configure()
  17. buttons := input.NewAnalogButtons(machine.A1)
  18. bzrPin := machine.A0
  19. bzrPin.Configure(machine.PinConfig{Mode: machine.PinOutput})
  20. bzr := buzzer.New(bzrPin)
  22. escornabot := bot.New(&motors, &buttons, &bzr, 542)
  24. escornabot.Loop()
  25. }
